\begin{recette}{Salade de Betterave crue à la Coriandre}
  \emph{Pour 2 personnes,	Préparation 15\,min }
  \begin{ingrédients}
    1 betterave crue « de belle taille » \sep
    $\frac{1}{2}$ orange\sep
    1\,c.c. de gingembre en poudre\sep
    1\,c.c. de cannelle en poudre\sep
    2\,c.s. de coriandre fraîche ciselée\sep
    1\,c.s. d’huile d’olive\sep
    (1\,c.c. de graine de sésame)\sep
    Sel, Poivre
  \end{ingredients}
  \recetteSection{Recette}
	\begin{enumerate}
  \item Épluchez et râpez la betterave. Pressez la demi-orange et versez le jus sur la betterave râpée. Ajoutez les épices, salez, poivrez.
  \item Incorporez ensuite la coriandre ciselée, arrosez d’huile d’olive et mélangez.
  \item Faites griller les graines de sésame à sec dans une poêle anti-adhésive et parsemez-en la salade avant de servir.
  \end{enumerate}
\end{recette}
