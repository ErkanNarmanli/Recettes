\begin{recette}{Purée améliorée}
  \emph{Pour 2\,personnes, Préparation 20\,min, Cuisson 50\,min}
  \begin{ingrédients}
    400\,g de pommes de terre\sep
    2\,oignons\sep
    3\,c.s. de crème fraîche\sep
    Lait\sep
    Thym
  \end{ingredients}
  \recetteSection{Recette}
	\begin{enumerate}
  \item Faites cuire les pommes de terre à l'eau jusqu'à ce qu'elles soient tendres (30 à 50\,min selon leur taille). Faites revenir les oignons à la poêle jusqu'à ce qu'ils caramélisent.
  \item Sortez les pommes-de-terre de l'eau, pelez-les et coupez-les en dés grossiers.
  \item Dans un saladier passez les pommes-de-terre, les oignons et la crème au presse-purée.
  \item Ajustez l'onctuosité avec le lait. Saupoudrez du thym à votre convenance, salez, poivrez.
  \end{enumerate}
\end{recette}
